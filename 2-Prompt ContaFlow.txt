## 🗄️ ESQUEMA DE BASE DE DATOS (Supabase/PostgreSQL)

Ejecuta esta migración en `supabase/migrations/001_initial_schema.sql`:

```sql
-- Habilitar extensiones necesarias
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- ENUM para niveles educativos
CREATE TYPE nivel_educativo AS ENUM (
  'basica_elemental',
  'basica_media', 
  'basica_superior',
  'bachillerato'
);

-- ENUM para grados
CREATE TYPE grado_escolar AS ENUM (
  '2do_egb', '3ro_egb', '4to_egb',
  '5to_egb', '6to_egb', '7mo_egb',
  '8vo_egb', '9no_egb', '10mo_egb',
  '1ro_bgu', '2do_bgu', '3ro_bgu'
);

-- Tabla: Perfiles de usuario (padres)
CREATE TABLE perfiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  nombre_completo TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  avatar_url TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Tabla: Estudiantes (hijos)
CREATE TABLE estudiantes (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  perfil_id UUID REFERENCES perfiles(id) ON DELETE CASCADE,
  nombre TEXT NOT NULL,
  edad INTEGER CHECK (edad >= 5 AND edad <= 18),
  grado grado_escolar NOT NULL,
  nivel nivel_educativo NOT NULL,
  avatar_id TEXT DEFAULT 'avatar_1',
  estrellas_totales INTEGER DEFAULT 0,
  racha_dias INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Tabla: Asignaturas
CREATE TABLE asignaturas (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  nombre TEXT NOT NULL,
  icono TEXT NOT NULL,
  color TEXT NOT NULL,
  descripcion TEXT,
  orden INTEGER DEFAULT 0
);

-- Tabla: Bloques curriculares
CREATE TABLE bloques_curriculares (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  asignatura_id UUID REFERENCES asignaturas(id) ON DELETE CASCADE,
  grado grado_escolar NOT NULL,
  titulo TEXT NOT NULL,
  descripcion TEXT,
  contenido_base TEXT,
  orden INTEGER DEFAULT 0
);

-- Tabla: Progreso del estudiante
CREATE TABLE progreso (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  estudiante_id UUID REFERENCES estudiantes(id) ON DELETE CASCADE,
  bloque_id UUID REFERENCES bloques_curriculares(id) ON DELETE CASCADE,
  completado BOOLEAN DEFAULT FALSE,
  estrellas_ganadas INTEGER DEFAULT 0 CHECK (estrellas_ganadas <= 3),
  tiempo_estudio INTEGER DEFAULT 0,
  ultima_sesion TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(estudiante_id, bloque_id)
);

-- Tabla: Historial de chat
CREATE TABLE historial_chat (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  estudiante_id UUID REFERENCES estudiantes(id) ON DELETE CASCADE,
  bloque_id UUID REFERENCES bloques_curriculares(id),
  rol TEXT NOT NULL CHECK (rol IN ('user', 'assistant')),
  mensaje TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Índices para rendimiento
CREATE INDEX idx_estudiantes_perfil ON estudiantes(perfil_id);
CREATE INDEX idx_bloques_grado ON bloques_curriculares(grado);
CREATE INDEX idx_progreso_estudiante ON progreso(estudiante_id);
CREATE INDEX idx_chat_estudiante ON historial_chat(estudiante_id);

-- RLS (Row Level Security)
ALTER TABLE perfiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE estudiantes ENABLE ROW LEVEL SECURITY;
ALTER TABLE progreso ENABLE ROW LEVEL SECURITY;
ALTER TABLE historial_chat ENABLE ROW LEVEL SECURITY;

-- Políticas RLS
CREATE POLICY "Usuarios ven su propio perfil" ON perfiles
  FOR ALL USING (auth.uid() = id);

CREATE POLICY "Usuarios ven sus estudiantes" ON estudiantes
  FOR ALL USING (perfil_id = auth.uid());

CREATE POLICY "Usuarios ven progreso de sus estudiantes" ON progreso
  FOR ALL USING (
    estudiante_id IN (
      SELECT id FROM estudiantes WHERE perfil_id = auth.uid()
    )
  );

CREATE POLICY "Usuarios ven chat de sus estudiantes" ON historial_chat
  FOR ALL USING (
    estudiante_id IN (
      SELECT id FROM estudiantes WHERE perfil_id = auth.uid()
    )
  );

-- Datos iniciales: Asignaturas
INSERT INTO asignaturas (nombre, icono, color, descripcion, orden) VALUES
  ('Matemáticas', 'calculator', '#3B82F6', 'Números, operaciones y geometría', 1),
  ('Lengua y Literatura', 'book-open', '#10B981', 'Lectura, escritura y comunicación', 2),
  ('Ciencias Naturales', 'flask-conical', '#8B5CF6', 'Naturaleza y medio ambiente', 3),
  ('Estudios Sociales', 'globe', '#F59E0B', 'Historia, geografía y civismo', 4);
```

## 🔌 API ROUTES

### Chat con IA (src/app/api/chat/route.ts)

```typescript
import { NextRequest, NextResponse } from "next/server"
import { createServerClient } from "@/lib/supabase/server"

// System prompt del tutor IA
const SYSTEM_PROMPT = `Eres "Profe Edu", un Psicólogo Educativo y Pedagogo ecuatoriano con 30 años de experiencia. Tu metodología combina refuerzo positivo y aprendizaje guiado (Socrático).

REGLAS ESTRICTAS:
1. NUNCA des la respuesta directa. Guía con preguntas.
2. Usa un tono cálido, paciente y entusiasta.
3. Adapta tu lenguaje a la edad del estudiante.
4. Si el estudiante falla, responde: "¡Buen intento! Veamos esto de otra forma..."
5. Celebra cada logro: "¡Excelente! ¡Eso merece una estrella! ⭐"
6. Usa ejemplos cotidianos de Ecuador cuando sea posible.
7. Mantén respuestas breves (máximo 3 párrafos).
8. Incluye emojis ocasionales para hacer la conversación amigable.

CONTEXTO DEL ESTUDIANTE:
- Nombre: {{nombre}}
- Edad: {{edad}} años
- Grado: {{grado}}
- Tema actual: {{tema}}

Inicia siempre saludando por el nombre y preguntando cómo se siente hoy.`

export async function POST(request: NextRequest) {
  try {
    const supabase = createServerClient()
    const { mensaje, estudiante_id, bloque_id } = await request.json()

    // Obtener datos del estudiante
    const { data: estudiante } = await supabase
      .from("estudiantes")
      .select("nombre, edad, grado")
      .eq("id", estudiante_id)
      .single()

    // Obtener contexto del bloque
    const { data: bloque } = await supabase
      .from("bloques_curriculares")
      .select("titulo, contenido_base")
      .eq("id", bloque_id)
      .single()

    // Obtener historial de chat (últimos 10 mensajes)
    const { data: historial } = await supabase
      .from("historial_chat")
      .select("rol, mensaje")
      .eq("estudiante_id", estudiante_id)
      .eq("bloque_id", bloque_id)
      .order("created_at", { ascending: true })
      .limit(10)

    // Construir prompt personalizado
    const promptPersonalizado = SYSTEM_PROMPT
      .replace("{{nombre}}", estudiante?.nombre || "Estudiante")
      .replace("{{edad}}", String(estudiante?.edad || 10))
      .replace("{{grado}}", estudiante?.grado || "")
      .replace("{{tema}}", bloque?.titulo || "tema general")

    // Aquí integrarías tu proveedor de IA preferido
    // Por ahora, retornamos estructura de ejemplo
    
    // Guardar mensaje del usuario
    await supabase.from("historial_chat").insert({
      estudiante_id,
      bloque_id,
      rol: "user",
      mensaje,
    })

    // TODO: Llamar a API de IA (OpenAI, Anthropic, etc.)
    const respuestaIA = "¡Hola! Soy el Profe Edu. ¿Cómo te sientes hoy?"

    // Guardar respuesta de IA
    await supabase.from("historial_chat").insert({
      estudiante_id,
      bloque_id,
      rol: "assistant",
      mensaje: respuestaIA,
    })

    return NextResponse.json({ 
      respuesta: respuestaIA,
      success: true 
    })

  } catch (error) {
    console.error("Error en chat:", error)
    return NextResponse.json(
      { error: "Error procesando mensaje" },
      { status: 500 }
    )
  }
}
```
